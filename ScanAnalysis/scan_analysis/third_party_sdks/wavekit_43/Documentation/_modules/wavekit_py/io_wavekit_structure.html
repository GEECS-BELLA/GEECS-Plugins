    """Class uint2D
    
    unsigned int (X,Y) couple
    :attribute X: X value
    :type X: uint
    :attribute Y: Y value
    :type Y: uint
    """
    _fields_ = [
        ('X', ctypes.c_uint),
        ('Y', ctypes.c_uint)
        ]
    def __init__(
        self,
        X, 
        Y
        ):
        super(uint2D, self).__init__(X,Y)
    def __str__(self):
        return (
            'uint2D Object : '+
            'X='+str(self.X)+'; '+
            'Y='+str(self.Y)
            )

    """Class dimensions
    
    :attribute size: size of the grid
    :type size: uint2D
    :attribute steps: spacing of the grid
    :type steps: float2D
    """
    _fields_ = [
        ('size', uint2D),
        ('steps', float2D)
        ]
    def __init__(
        self, 
        size, 
        steps
        ):

        return (
            'dimensions Object : '+
            'size='+str(self.size)+'; '+
